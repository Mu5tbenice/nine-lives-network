--- GAMEPLAY ENGINE --- def key_press(self, event): key = event.keysym.lower() if key in self.keys: self.keys[key] = True def key_release(self, event): key = event.keysym.lower() if key in self.keys: self.keys[key] = False def get_angle_from_vector(self, dx, dy): if dx > 0.3 and abs(dy) < 0.3: return "090" if dx < -0.3 and abs(dy) < 0.3: return "270" if dy > 0.3 and abs(dx) < 0.3: return "000" if dy < -0.3 and abs(dx) < 0.3: return "180" if dx > 0 and dy > 0: return "045" if dx > 0 and dy < 0: return "135" if dx < 0 and dy > 0: return "315" if dx < 0 and dy < 0: return "225" return self.current_angle.get() def start_game(self): self.is_wandering = False self.is_spinning = False self.in_game = True self.hp = 100 self.score = 0 self.enemies = [] self.projectiles = [] self.shoot_cooldown = 0 self.spawn_timer = 30 self.spawn_interval = 60 # AUTO-SHOOTER LOGIC if self.shoot_cooldown > 0: self.shoot_cooldown -= 1 # Find nearest enemy nearest_enemy = None min_dist = float('inf') for e in self.enemies: dist = math.hypot(e['x'] - self.char_x, e['y'] - self.char_y) if dist < min_dist: min_dist = dist nearest_enemy = e # Shoot projectile if ready and enemy is within 400px if self.shoot_cooldown <= 0 and nearest_enemy and min_dist < 400: edx = nearest_enemy['x'] - self.char_x edy = nearest_enemy['y'] - self.char_y dist = math.hypot(edx, edy) if dist > 0: proj_speed = 20 self.projectiles.append({ 'x': self.char_x, 'y': self.char_y, 'dx': (edx/dist) * proj_speed, 'dy': (edy/dist) * proj_speed, 'life': 50 # Dies after 50 frames if it misses }) self.shoot_cooldown = 12 # Fire rate (lower is faster) # 3. PROJECTILE UPDATES for p in self.projectiles[:]: p['x'] += p['dx'] p['y'] += p['dy'] p['life'] -= 1 hit = False for e in self.enemies[:]: if math.hypot(e['x'] - p['x'], e['y'] - p['y']) < 40: # Projectile hit radius e['hp'] -= 35 # Damage per shot (3 hits to kill) hit = True break if hit or p['life'] <= 0: if p in self.projectiles: self.projectiles.remove(p) # 4. ENEMY AI & COMBAT (Swarm logic) for e in self.enemies[:]: edx = self.char_x - e['x'] edy = self.char_y - e['y'] dist = math.hypot(edx, edy) # Follow Player if dist > 0: # Enemies move slightly slower than the player's set speed e['x'] += (edx/dist) * (max(2, speed * 0.5)) e['y'] += (edy/dist) * (max(2, speed * 0.5)) e['angle'] = self.get_angle_from_vector(edx/dist, edy/dist) # Player takes melee damage from cats if dist < 45: self.hp -= 1.5 # Drain health while touching # Enemy death if e['hp'] <= 0: self.score += 10 self.enemies.remove(e) # 5.
